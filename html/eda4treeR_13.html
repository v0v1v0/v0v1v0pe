Examples

library(car)
library(dae)
library(dplyr)
library(emmeans)
library(ggplot2)
library(lmerTest)
library(magrittr)
library(predictmeans)

data(DataExam2.1)
# Pg. 22
fmtab2.3  <- lm(formula = dbh ~ seedlot, data = DataExam2.1)
# Pg. 23
anova(fmtab2.3)

# Pg. 23
emmeans(object = fmtab2.3, specs = ~ seedlot)
emmip(object = fmtab2.3, formula = ~ seedlot) +
  theme_classic()
